1//! Aggregation operators for GROUP BY operations
2
3use crate::error::{Result, VectrillError};
4use arrow::array::ArrayRef;

6use arrow::record_batch::RecordBatch as ArrowRecordBatch;
7use std::collections::HashMap;
8use std::sync::Arc;
9
10use super::Operator;

12/// Aggregation function types
13#[derive(Debug, Clone, Copy, PartialEq)]
14pub enum AggregateFunction {
15    Sum,
16    Avg,
17    Min,
18    Max,
19    Count,
20}
21
22/// Aggregation operator for GROUP BY operations
23pub struct AggregateOperator {
24    /// Group by column names
25    group_by: Vec<String>,
26    /// Aggregate specifications: (column name, aggregate function, output column name)
27    aggregates: Vec<(String, AggregateFunction, String)>,
28    /// Aggregation state
29    state: Option<AggregationState>,
30}
31
32#[derive(Debug)]
33struct AggregationState {
34    /// Group key to aggregate values mapping
35    groups: HashMap<String, Vec<ArrayRef>>,
36    /// Schema of the output
37    schema: SchemaRef,
38}
39
40impl AggregateOperator {
41    pub fn new(
42        group_by: Vec<String>,
43        aggregates: Vec<(String, AggregateFunction, String)>,
44    ) -> Self {
45        Self {
46            group_by,
47            aggregates,
48            state: None,
49        }
50    }
51
52    fn create_output_schema(&self, input_schema: &Schema) -> Result<SchemaRef> {
53        let mut fields = Vec::new();
54
55        // Add group by columns
56        for col_name in &self.group_by {
57            if let Ok(field) = input_schema.field_with_name(col_name) {
58                fields.push(field.clone());
59            } else {
60                return Err(VectrillError::InvalidSchema(format!(
61                    "Group by column '{}' not found in schema",
62                    col_name
63                )));
64            }
65        }
66
67        // Add aggregate columns
68        for (_, _, output_name) in &self.aggregates {
69            // For simplicity, use Float64 for all aggregates except Count which is Int64
70            fields.push(Field::new(output_name, DataType::Float64, true));
71        }
72
73        Ok(Arc::new(Schema::new(fields)))
74    }
75}
76
77impl Operator for AggregateOperator {
78    fn process(&mut self, batch: crate::RecordBatch) -> Result<crate::RecordBatch> {
79        let arrow_batch = batch;
80
81        // Initialize state if not already done
82        if self.state.is_none() {
83            let schema = self.create_output_schema(arrow_batch.schema().as_ref())?;
84            self.state = Some(AggregationState {
85                groups: HashMap::new(),
86                schema,
87            });
88        }
89
90        let state = self.state.as_mut().unwrap();
91
92        // Process each row and update aggregation state
93        let num_rows = arrow_batch.num_rows();
94
95        for _i in 0..num_rows {
96            // Build group key
97            let mut group_key = String::new();
98            for (idx, col_name) in self.group_by.iter().enumerate() {
99                if idx > 0 {
100                    group_key.push('|');
101                }
102
103                let col = arrow_batch.column_by_name(col_name).ok_or_else(|| {
104                    VectrillError::InvalidSchema(format!("Column '{}' not found", col_name))
105                })?;
106
107                // For simplicity, convert column value to string
108                group_key.push_str(&format!("{:?}", col));
109            }
110
111            // Store row data for aggregation
112            state
113                .groups
114                .entry(group_key)
115                .or_insert_with(Vec::new)
116                .extend(arrow_batch.columns().iter().cloned());
117        }
118
119        // Return empty batch for now (aggregation happens on flush)
120        Ok(ArrowRecordBatch::new_empty(state.schema.clone()))
121    }
122
123    fn flush(&mut self) -> Result<Vec<crate::RecordBatch>> {
124        if let Some(state) = self.state.take() {
125            if state.groups.is_empty() {
126                return Ok(vec![]);
127            }
128
129            // Compute final aggregates for each group
130            let mut output_rows = Vec::new();
131
132            for (group_key, rows) in state.groups {
133                // Parse group key to get group by values
134                let group_values: Vec<String> =
135                    group_key.split('|').map(|s| s.to_string()).collect();
136
137                // Compute aggregates
138                let mut aggregate_values = Vec::new();
139
140                for (_col_name, agg_fn, _) in &self.aggregates {
141                    // For simplicity, use a placeholder implementation
142                    match agg_fn {
143                        AggregateFunction::Sum => {
144                            let sum = 0.0_f64;
145                            aggregate_values.push(sum);
146                        }
147                        AggregateFunction::Avg => {
148                            let avg = 0.0_f64;
149                            aggregate_values.push(avg);
150                        }
151                        AggregateFunction::Min => {
152                            let min = 0.0_f64;
153                            aggregate_values.push(min);
154                        }
155                        AggregateFunction::Max => {
156                            let max = 0.0_f64;
157                            aggregate_values.push(max);
158                        }
159                        AggregateFunction::Count => {
160                            let count = rows.len() as i64;
161                            aggregate_values.push(count as f64);
162                        }
163                    }
164                }
165
166                output_rows.push((group_values, aggregate_values));
167            }
168
169            // Create output batch
170            // This is a simplified implementation - real implementation would build proper Arrow arrays
171            Ok(vec![ArrowRecordBatch::new_empty(state.schema)])
172        } else {
173            Ok(vec![])
174        }
175    }
176}
