6/// Binary operators
7#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
8pub enum Operator {
9    // Comparison operators
10    Eq,
11    NotEq,
12    Lt,
13    LtEq,
14    Gt,
15    GtEq,
16
17    // Arithmetic operators
18    Add,
19    Sub,
20    Mul,
21    Div,
22    Mod,
23
24    // Boolean operators
25    And,
26    Or,
27
28    // String operators
29    Like,
30    ILike,
31    NotLike,
32    NotILike,
33
34    // Array operators
35    In,
36    NotIn,
37
38    // Null operators
39    IsNull,
40    IsNotNull,
41}
42
43impl fmt::Display for Operator {
44    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
45        match self {
46            Operator::Eq => write!(f, "="),
47            Operator::NotEq => write!(f, "!="),
48            Operator::Lt => write!(f, "<"),
49            Operator::LtEq => write!(f, "<="),
50            Operator::Gt => write!(f, ">"),
51            Operator::GtEq => write!(f, ">="),
52            Operator::Add => write!(f, "+"),
53            Operator::Sub => write!(f, "-"),
54            Operator::Mul => write!(f, "*"),
55            Operator::Div => write!(f, "/"),
56            Operator::Mod => write!(f, "%"),
57            Operator::And => write!(f, "AND"),
58            Operator::Or => write!(f, "OR"),
59            Operator::Like => write!(f, "LIKE"),
60            Operator::ILike => write!(f, "ILIKE"),
61            Operator::NotLike => write!(f, "NOT LIKE"),
62            Operator::NotILike => write!(f, "NOT ILIKE"),
63            Operator::In => write!(f, "IN"),
64            Operator::NotIn => write!(f, "NOT IN"),
65            Operator::IsNull => write!(f, "IS NULL"),
66            Operator::IsNotNull => write!(f, "IS NOT NULL"),
67        }
68    }
69}

91/// Operator precedence levels
92#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
93pub enum Precedence {
94    Lowest = 0,
95    Or = 1,
96    And = 2,
97    Comparison = 3,
98    AddSub = 4,
99    MulDiv = 5,
100    Unary = 6,
101    Highest = 7,
102}
103
104impl Operator {
105    /// Get the precedence level of this operator
106    pub fn precedence(&self) -> Precedence {
107        match self {
108            Operator::Or => Precedence::Or,
109            Operator::And => Precedence::And,
110            Operator::Eq
111            | Operator::NotEq
112            | Operator::Lt
113            | Operator::LtEq
114            | Operator::Gt
115            | Operator::GtEq => Precedence::Comparison,
116            Operator::Add | Operator::Sub => Precedence::AddSub,
117            Operator::Mul | Operator::Div | Operator::Mod => Precedence::MulDiv,
118            Operator::Like | Operator::ILike | Operator::NotLike | Operator::NotILike => {
119                Precedence::Comparison
120            }
121            Operator::In | Operator::NotIn => Precedence::Comparison,
122            Operator::IsNull | Operator::IsNotNull => Precedence::Comparison,
123        }
124    }
125
126    /// Check if this operator is left-associative
127    pub fn is_left_associative(&self) -> bool {
128        match self {
129            Operator::Add | Operator::Sub | Operator::Mul | Operator::Div | Operator::Mod => true,
130            Operator::And | Operator::Or => true,
131            Operator::Eq
132            | Operator::NotEq
133            | Operator::Lt
134            | Operator::LtEq
135            | Operator::Gt
136            | Operator::GtEq => true,
137            Operator::Like | Operator::ILike | Operator::NotLike | Operator::NotILike => true,
138            Operator::In | Operator::NotIn => true,
139            Operator::IsNull | Operator::IsNotNull => true,
140        }
141    }
142
143    /// Check if this operator is a comparison operator
144    pub fn is_comparison(&self) -> bool {
145        matches!(
146            self,
147            Operator::Eq
148                | Operator::NotEq
149                | Operator::Lt
150                | Operator::LtEq
151                | Operator::Gt
152                | Operator::GtEq
153                | Operator::Like
154                | Operator::ILike
155                | Operator::NotLike
156                | Operator::NotILike
157                | Operator::In
158                | Operator::NotIn
159                | Operator::IsNull
160                | Operator::IsNotNull
161        )
162    }
163
164    /// Check if this operator is arithmetic
165    pub fn is_arithmetic(&self) -> bool {
166        matches!(
167            self,
168            Operator::Add | Operator::Sub | Operator::Mul | Operator::Div | Operator::Mod
169        )
170    }
171
172    /// Check if this operator is logical
173    pub fn is_logical(&self) -> bool {
174        matches!(self, Operator::And | Operator::Or)
175    }
176}

178impl UnaryOp {
179    /// Get the precedence level of this unary operator
180    pub fn precedence(&self) -> Precedence {
181        Precedence::Unary
182    }
183}

185/// Function to map Python AST operator strings to our Operator enum
186pub fn map_python_operator(op_str: &str) -> Option<Operator> {
187    match op_str {
188        "Eq" => Some(Operator::Eq),
189        "NotEq" => Some(Operator::NotEq),
190        "Lt" => Some(Operator::Lt),
191        "LtE" => Some(Operator::LtEq),
192        "Gt" => Some(Operator::Gt),
193        "GtE" => Some(Operator::GtEq),
194        "Add" => Some(Operator::Add),
195        "Sub" => Some(Operator::Sub),
196        "Mult" => Some(Operator::Mul),
197        "Div" => Some(Operator::Div),
198        "Mod" => Some(Operator::Mod),
199        "And" => Some(Operator::And),
200        "Or" => Some(Operator::Or),
201        _ => None,
202    }
203}
204
205/// Function to map Python boolean operator strings to our Operator enum
206pub fn map_python_bool_op(op_str: &str) -> Option<Operator> {
207    match op_str {
208        "And" => Some(Operator::And),
209        "Or" => Some(Operator::Or),
210        _ => None,
211    }
212}

214/// Function to map Python unary operator strings to our UnaryOp enum
215pub fn map_python_unary_op(op_str: &str) -> Option<UnaryOp> {
216    match op_str {
217        "Not" => Some(UnaryOp::Not),
218        "USub" => Some(UnaryOp::Neg),
219        _ => None,
220    }

244        assert!(Operator::And.is_logical());

256    }
